cmake_minimum_required(VERSION 3.15)
project(CameraDisplays)

set(CMAKE_CXX_STANDARD 14)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Find OpenCV
find_package(OpenCV REQUIRED)
include_directories(${OpenCV_INCLUDE_DIRS})

# Find Pylon SDK
set(PYLON_ROOT "$ENV{PYLON_ROOT}" CACHE PATH "Pylon SDK root directory")
if(NOT PYLON_ROOT)
    if(EXISTS "C:/Program Files/Basler/pylon 7")
        set(PYLON_ROOT "C:/Program Files/Basler/pylon 7")
    elseif(EXISTS "C:/Program Files/Basler/pylon 6")
        set(PYLON_ROOT "C:/Program Files/Basler/pylon 6")
    elseif(EXISTS "C:/Program Files (x86)/Basler/pylon 7")
        set(PYLON_ROOT "C:/Program Files (x86)/Basler/pylon 7")
    elseif(EXISTS "C:/Program Files (x86)/Basler/pylon 6")
        set(PYLON_ROOT "C:/Program Files (x86)/Basler/pylon 6")
    endif()
endif()

if(PYLON_ROOT)
    include_directories("${PYLON_ROOT}/include")
    link_directories("${PYLON_ROOT}/lib/x64")
    message(STATUS "Pylon SDK found at: ${PYLON_ROOT}")
else()
    message(FATAL_ERROR "Pylon SDK not found. Please set PYLON_ROOT environment variable.")
endif()

# Find Spinnaker SDK
set(SPINNAKER_ROOT "$ENV{SPINNAKER_ROOT}" CACHE PATH "Spinnaker SDK root directory")
if(NOT SPINNAKER_ROOT)
    if(EXISTS "C:/Program Files/FLIR Systems/Spinnaker")
        set(SPINNAKER_ROOT "C:/Program Files/FLIR Systems/Spinnaker")
    elseif(EXISTS "C:/Program Files (x86)/FLIR Systems/Spinnaker")
        set(SPINNAKER_ROOT "C:/Program Files (x86)/FLIR Systems/Spinnaker")
    elseif(EXISTS "C:/Program Files/Point Grey Research/Spinnaker")
        set(SPINNAKER_ROOT "C:/Program Files/Point Grey Research/Spinnaker")
    endif()
endif()

# Include parent directory for HardwareTriggerConfiguration.h
include_directories("${CMAKE_SOURCE_DIR}/..")

# Pylon libraries
set(PYLON_LIBS
    PylonBase_v7_4.lib
    PylonC_v7_3.lib
    PylonUtility_v7_4.lib
    GenApi_MD_VC141_v3_1_Basler_pylon.lib
    GCBase_MD_VC141_v3_1_Basler_pylon.lib
)

# === Bottom Camera Display (Single Basler) ===
add_executable(bottom_camera_display bottom_camera_display.cpp)
target_link_libraries(bottom_camera_display ${OpenCV_LIBS} ${PYLON_LIBS})

# Copy Pylon DLLs for bottom_camera_display
if(EXISTS "${PYLON_ROOT}/Runtime/x64")
    add_custom_command(TARGET bottom_camera_display POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_directory
        "${PYLON_ROOT}/Runtime/x64"
        $<TARGET_FILE_DIR:bottom_camera_display>
    )
endif()

# === Dual Basler Display (Two Basler cameras) ===
add_executable(dual_basler_display dual_basler_display.cpp)
target_link_libraries(dual_basler_display ${OpenCV_LIBS} ${PYLON_LIBS})

# Copy Pylon DLLs for dual_basler_display
if(EXISTS "${PYLON_ROOT}/Runtime/x64")
    add_custom_command(TARGET dual_basler_display POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_directory
        "${PYLON_ROOT}/Runtime/x64"
        $<TARGET_FILE_DIR:dual_basler_display>
    )
endif()

# === Dual Camera Display (Basler + Spinnaker) ===
if(SPINNAKER_ROOT)
    message(STATUS "Spinnaker SDK found at: ${SPINNAKER_ROOT}")
    
    include_directories("${SPINNAKER_ROOT}/include")
    link_directories("${SPINNAKER_ROOT}/lib64/vs2015")
    
    # Spinnaker libraries
    set(SPINNAKER_LIBS
        Spinnaker_v140.lib
    )
    
    add_executable(dual_camera_display dual_camera_display.cpp)
    target_link_libraries(dual_camera_display 
        ${OpenCV_LIBS} 
        ${PYLON_LIBS}
        ${SPINNAKER_LIBS}
    )
    
    # Copy DLLs for dual_camera_display
    if(EXISTS "${PYLON_ROOT}/Runtime/x64")
        add_custom_command(TARGET dual_camera_display POST_BUILD
            COMMAND ${CMAKE_COMMAND} -E copy_directory
            "${PYLON_ROOT}/Runtime/x64"
            $<TARGET_FILE_DIR:dual_camera_display>
        )
    endif()
    
    if(EXISTS "${SPINNAKER_ROOT}/bin64/vs2015")
        add_custom_command(TARGET dual_camera_display POST_BUILD
            COMMAND ${CMAKE_COMMAND} -E copy_directory
            "${SPINNAKER_ROOT}/bin64/vs2015"
            $<TARGET_FILE_DIR:dual_camera_display>
        )
    endif()
    
    message(STATUS "Dual camera display will be built")
else()
    message(WARNING "Spinnaker SDK not found. Only bottom_camera_display will be built.")
    message(WARNING "To build dual_camera_display, install Spinnaker SDK and set SPINNAKER_ROOT environment variable.")
endif()
